Sub submit(sender As Object, e As EventArgs)

   lbl1.Text="Page is valid."
Else

End If
End Sub
</script>

<asp:TextBox id="tbox1" runat="server" />

<br><br>
<asp:Label id="lbl1" runat="server" />
